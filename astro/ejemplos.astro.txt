# Factorial Recursivo
factorial = (x) ->
  if x == 0
  then 1
  else x * factorial (x-1)

# Factorial Pocedural
factorial = (x) ->
  r = 1
  for i in [1..x]
    r = r*i

# Factorial OOP
factorial = (x) ->
  r = 1
  x.count (i) ->
    r = r*i  # r es global a menos que exista una local

# Factorial Super Funcional
factorial = (x) ->
  [1..x].fold (a,b)->a*b
  [1..x].fold %op(*) # Talves asÃ­?
